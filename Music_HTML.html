    let currentIndex = 0;

    let isPlaying = false;   // track if audio is currently playing

    let isShuffle = false;

    let hasUserInteracted = false; // For iOS Safari user-gesture policies


    let audioElement;

    let videoElement;

    let progressBar;

    /*****************************************************

     *   UTILITY FUNCTIONS

     *****************************************************/

    function formatTime(seconds) {

      const mins = Math.floor(seconds / 60);

      const secs = Math.floor(seconds % 60);

      return `${mins}:${secs < 10 ? "0" : ""}${secs}`;

    }


    function scrollToVideo() {

      const videoSection = document.getElementById("videoSection");

      videoSection.scrollIntoView({ behavior: "smooth" });

    }


    /*****************************************************

     *   AUDIO PLAYER LOGIC

     *****************************************************/

    function loadTrack(index) {

      currentIndex = index;

      const track = tracks[currentIndex];


      // Update audio source & cover art

      audioElement.src = track.audio;

      if (track.cover) {

        currentCover.src = track.cover;

      } else {

        currentCover.src = "https://via.placeholder.com/200/333333/ffffff?text=No+Cover";

      }


      // Reset 'isPlaying' to false until user triggers play

      isPlaying = false;

      playPauseBtn.textContent = "▶"; // show 'Play' symbol


      // Update duration after metadata loads

      audioElement.addEventListener("loadedmetadata", () => {

        durationDisplay.textContent = formatTime(audioElement.duration);

      }, { once: true });

    }


    // When user manually picks Next

    function nextTrack() {

      if (isShuffle) {

        let randomIndex;

        do {

          randomIndex = Math.floor(Math.random() * tracks.length);

        } while (randomIndex === currentIndex);

        currentIndex = randomIndex;

      } else {

        currentIndex++;

        if (currentIndex >= tracks.length) {

          currentIndex = 0;

        }

      }

      loadTrack(currentIndex);


      // Attempt autoplay if user has interacted

      audioElement.load();

      if (hasUserInteracted) {

        audioElement.play().catch(err => console.log("iOS user gesture required:", err));

      }

    }


    // When user manually picks Prev

    function prevTrack() {

      if (isShuffle) {

        let randomIndex;

        do {

          randomIndex = Math.floor(Math.random() * tracks.length);

        } while (randomIndex === currentIndex);

        currentIndex = randomIndex;

      } else {

        currentIndex--;

        if (currentIndex < 0) {

          currentIndex = tracks.length - 1;

        }

      }

      loadTrack(currentIndex);


      // Attempt autoplay if user has interacted

      audioElement.load();

      if (hasUserInteracted) {

        audioElement.play().catch(err => console.log("iOS user gesture required:", err));

      }

    }


    function toggleShuffle() {

      isShuffle = !isShuffle;

      shuffleBtn.textContent = `🔀 ${isShuffle ? "On" : "Off"}`;

    }


    // Toggle Play/Pause from the main button

    function togglePlayPause() {

      if (!hasUserInteracted) {

        hasUserInteracted = true; // The user tapped the play/pause button

      }


      if (isPlaying) {

        audioElement.pause();

      } else {

        audioElement.play().catch(err => {

          console.log("Playback blocked, iOS user gesture required:", err);

        });

      }

    }


    function updateProgress() {

      if (!isNaN(audioElement.duration)) {

        const progress = (audioElement.currentTime / audioElement.duration) * 100;

        progressBar.value = progress;

        currentTimeDisplay.textContent = formatTime(audioElement.currentTime);

      }

    }


    function seekAudio() {

      const seekTime = (progressBar.value / 100) * audioElement.duration;

      audioElement.currentTime = seekTime;

    }


    /*****************************************************

     *   LYRIC VIDEO LOGIC

     *****************************************************/

    function playLyricVideo(videoUrl) {

      if (!videoUrl) return;

      // The user clicked the “Lyric Video” button, so it’s a valid gesture

      hasUserInteracted = true;


      videoElement.src = videoUrl;

      videoElement.play().catch(err => {

        console.log("iOS gesture required for video playback:", err);

      });

      scrollToVideo();

    }


    /*****************************************************

     *   BUILD SONG LIST

     *****************************************************/

    function buildSongList() {

      const songListDiv = document.getElementById("songList");


      tracks.forEach((track, index) => {

        const itemDiv = document.createElement("div");

        itemDiv.className = "song-item";


        // Thumbnail if there's cover art, else placeholder

        const thumbnail = document.createElement("img");

        thumbnail.src = track.cover

          ? track.cover

          : "https://via.placeholder.com/60/333333/ffffff?text=No+Cover";

        itemDiv.appendChild(thumbnail);


        // Song info

        const infoDiv = document.createElement("div");

        infoDiv.className = "song-info";

        const titleEl = document.createElement("h3");

        titleEl.textContent = track.title;

        infoDiv.appendChild(titleEl);


        itemDiv.appendChild(infoDiv);


        // If the song has a lyric video, add a "Lyric Video" button

        if (track.video) {

          const lyricBtn = document.createElement("button");

          lyricBtn.textContent = "Lyric Video";

          lyricBtn.className = "lyric-btn";

          lyricBtn.addEventListener("click", (event) => {

            // Prevent the itemDiv click from also firing

            event.stopPropagation();

            playLyricVideo(track.video);


            // Pause audio if playing

            audioElement.pause();

          });

          itemDiv.appendChild(lyricBtn);

        }


        // Entire .song-item is clickable -> loadTrack(index) + attempt to play audio

        itemDiv.addEventListener("click", () => {

          // The user tapped a song in the list, so we have a valid gesture

          hasUserInteracted = true;


          loadTrack(index);

          audioElement.load();

          audioElement.play().catch(err => {

            console.log("Playback blocked, iOS user gesture required:", err);

          });


          // Stop any playing video

          videoElement.pause();

          videoElement.src = "";

        });


        songListDiv.appendChild(itemDiv);

      });

    }

        playPauseBtn.textContent = "▶"; // play symbol

      // Load the first track's data (but do NOT attempt to autoplay)

      loadTrack(0);
